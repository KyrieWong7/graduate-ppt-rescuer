%% \speakbox{内容} — 蓝色边框，直接口播的内容
\newcommand{\speakbox}[1]{%
  \vspace{4pt}
  \noindent\fcolorbox{speakblue}{speakbg}{%

      {\normalsize #1}

  }
  \vspace{4pt}
}

      {\small\color{notegray} #1}
      \vspace{2pt}
    \end{minipage}
  }
  \vspace{4pt}
}
